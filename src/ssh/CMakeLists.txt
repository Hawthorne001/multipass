function(add_ssh_target TARGET_NAME)
  add_library(${TARGET_NAME} STATIC
    openssh_key_provider.cpp
    ssh_client_key_provider.cpp
    ssh_process.cpp
    ssh_session.cpp)

  target_link_libraries(${TARGET_NAME}
    fmt
    libssh
    utils
    Qt5::Core)
endfunction()

add_ssh_target(ssh)
if(MULTIPASS_ENABLE_TESTS)
  add_ssh_target(ssh_test)
endif()
